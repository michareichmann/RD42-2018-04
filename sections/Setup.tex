\subsection{Pixel Telescope}
\begin{frame}{Pixel Telescope}

	\vspace*{-7pt}\figc{3DTel1}{.48}{modular ETH beam telescope in pixel configuration}\vspace*{-10pt}
	
	\begin{itemize}\itemfill
		\item 4 tracking planes with trigger (fast-OR) and \SI{25}{\nano\second} time precision
		\item up to 3 DUT planes (any digital pixel detector)
		\item scintillator for precise trigger timing \ra \orderof{\SI{1}{\nano\second}}
		\item spatial resolution of \SI{\sim188x175}{\micro\meter} (multiple scattering)
	\end{itemize}

\end{frame}
