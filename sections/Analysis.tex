\subsection{Event Selection}
\begin{frame}{General Cuts}

	\begin{itemize}\itemfill
		\item Event Range
		\begin{itemize}\vspace*{3pt}
			\item discard first minute of the run \ra unstable beam conditions
		\end{itemize}
		\item Beam Interruptions
		\begin{itemize}\vspace*{3pt}
			\item beam sent to another area for \SI{5}{\second} every \SI{5}{\minute}
			\item exclude \SI{5}{\second} before and \SI{15}{\second} after the interruption
		\end{itemize}
		\item Tracks
		\begin{itemize}\vspace*{3pt}
			\item only events with exact one cluster in every tracking plane
		\end{itemize}
		\item Track $\upchi^2$
		\begin{itemize}\vspace*{3pt}
			\item exclude events with $\upchi^2 > \SI{90}{\%}$ quartile separately in x- and y-direction
		\end{itemize}
		\item Track Angle
		\begin{itemize}\vspace*{3pt}
			\item exclude events with $\upalpha > \SI{2}{^{\circ}}$ from mean angle
		\end{itemize}
		\item R-Hit
		\begin{itemize}\vspace*{3pt}
			\item discard events with distance of predicted hit position to cluster position \SI{>90}{\%} quartile
		\end{itemize}
	\end{itemize}
	
\end{frame}
